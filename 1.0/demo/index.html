1. 数据源为Arr

出发时间
已选择:

S.use('gallery/simulate-select/1.0/index, overlay', function (S, SimulateSelect, overlay) {

var select = new SimulateSelect({

selectNode: '#J_demo1',

options: [{text: '1', value: '1', disabled: true}, {text: '2', value: '2', disabled: false}],

isShowSelectText: false,

name: 'test',

multi: false,

isShowSelectBox: false,

isHideBoxBySelected: true

});

select.on("change", function(e){

S.Node.one("#J_demo1_value").html(select.value());

});

});
